	fun p -> 
		let f = p.snd in
		let z = f 0 in

	fun p ->
		let n = p.fst in

	fun n ->

	fun x -> fun p ->
		let n = p.fst in
		let f = p.snd in
		let g : j:int -> int[r.r > 0] = fun j ->
			(assume (j = 0); x) <>

let fold_left : f : (x1 : int -> x2 : int[x2.x2 > 0] -> int) -> 
                acc : int -> 
                xs : (x1 : int * (i : int -> int[r.r > 0])) -> 
                int =
  fun f -> fun acc -> fun xs -> 
    (assume xs.fst = 0;acc) <>
    (assume xs.fst > 0;
     let xs1 = tail xs in
     let x   = head xs in
     let acc1 = f acc x in
     let r   = fold_left f acc1 xs1 in
     r);;
