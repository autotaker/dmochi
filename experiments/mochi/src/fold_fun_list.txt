let head : p : (n : int * (i : int -> x : int -> int[r.1 <= - x + r])) -> x : int -> int[r. 1 <= - x + r] =
  fun p ->
    let f = p.snd in
    let z = f 0 in
    z;;

let tail : p: (n : int * (i : int -> x : int -> int[r. 1 <= - x + r])) -> 
            (m : int * (j : int -> y : int -> int[r.1 <= -y + r])) =
	fun p ->
		let n = p.fst in
		let f = p.snd in
		let g : j:int -> y : int -> int[r.1  <= - y + r ] = fun j -> let v = f (j + 1) in v in
		(n-1,g);;

let fold_right : h : (f : (x : int -> int[r. 1 <= - x + r]) -> g : (x : int -> int[r.r >= x]) -> x : int -> int[r. 1 <= -x + r]) -> 
                 xs : (n : int * (i : int -> x : int -> int[r. 1 <= - x + r])) ->
                 init : (x : int -> int[r.r >= x]) ->
                 x : int -> int[r. 1 <= - x + r; r. r >= x] =
  fun h -> fun xs -> fun init ->
    (assume xs.fst = 0; init) <>
    (assume xs.fst > 0; 
      let f = head xs in
      let xs1 = tail xs in
      let g = fold_right h xs1 init in
      let r = h f g in
      r);;

let compose : f : (x : int -> int[r. 1 <= - x + r]) -> g : (x : int -> int[r. r >= x]) -> x : int -> int[r. 1 <= -x + r] =
  fun f -> fun g -> fun x ->
    let x1 = g x in
    let x2 = f x1 in
    x2;;

let make_list : n : int -> (m : int[m. n >= 0] * (j : int -> y : int -> int[r. 1 <= -y + r])) =
  fun n ->
    (assume n = 0;
		 let f : j:int -> y : int -> int[r.1 <= -y + r] = 
       fun j -> (assume false; fun y -> y) in
		 (0,f)) <>
    (assume not (n = 0);
     let r = make_list (n - 1) in
     let x : m : int -> int[r. 1 <= - m + r] = fun m -> n + m in
		 let n = r.fst in
		 let f = r.snd in
		 let g : j : int -> y : int -> int[r. 1 <= -y + r] = fun j ->
			 (assume (j = 0); x) <>
			 (assume (not (j = 0)); 
			  let z = f (j - 1) in z) in
		(n+1,g));;
